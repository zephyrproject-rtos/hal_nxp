42 /*! @brief Register settings for Normal (non buffered) mode. */
43 const registerwritelist_t cMag3110ConfigNormal[] = {
44  /* Set Ouput Rate @10HZ (ODR = 2 and OSR = 32). */
45  {MAG3110_CTRL_REG1, MAG3110_CTRL_REG1_DR_ODR_2 | MAG3110_CTRL_REG1_OS_OSR_32,
46  MAG3110_CTRL_REG1_DR_MASK | MAG3110_CTRL_REG1_OS_MASK},
47  /* Set Auto Magnetic Sensor Reset. */
48  {MAG3110_CTRL_REG2, MAG3110_CTRL_REG2_MAG_RST_EN | MAG3110_CTRL_REG2_AUTO_MSRT_EN_EN | MAG3110_CTRL_REG2_RAW_RAW,
49  MAG3110_CTRL_REG2_MAG_RST_MASK | MAG3110_CTRL_REG2_AUTO_MSRT_EN_MASK | MAG3110_CTRL_REG2_RAW_MASK},
50  __END_WRITE_DATA__};

62 /*!
63  * @brief Main function
64  */
65 int main(void)
66 {
67  int32_t status;
68  uint8_t dataReady;
69  uint8_t data[MAG3110_DATA_SIZE];
70  mag3110_magdata_t rawData;
71 
72  ARM_DRIVER_I2C *I2Cdrv = &I2C_S_DRIVER; // Now using the shield.h value!!!
73  mag3110_i2c_sensorhandle_t mag3110Driver;
74 
75  BOARD_InitPins();
76  BOARD_BootClockRUN();
77  BOARD_InitDebugConsole();
78 
79  PRINTF("\r\n ISSDK MAG3110 sensor driver example demonstration with poll mode\r\n");
80 
81  /*! Initialize the I2C driver. */
82  status = I2Cdrv->Initialize(I2C_S_SIGNAL_EVENT);
83  if (ARM_DRIVER_OK != status)
84  {
85  PRINTF("\r\n I2C Initialization Failed\r\n");
86  return -1;
87  }
88 
89  /*! Set the I2C Power mode. */
90  status = I2Cdrv->PowerControl(ARM_POWER_FULL);
91  if (ARM_DRIVER_OK != status)
92  {
93  PRINTF("\r\n I2C Power Mode setting Failed\r\n");
94  return -1;
95  }
96 
97  /*! Set the I2C bus speed. */
98  status = I2Cdrv->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST);
99  if (ARM_DRIVER_OK != status)
100  {
101  PRINTF("\r\n I2C Control Mode setting Failed\r\n");
102  return -1;
103  }
104 
105  /*! Initialize MAG3110 sensor driver. */
106  status = MAG3110_I2C_Initialize(&mag3110Driver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, MAG3110_I2C_ADDR,
107  MAG3110_WHOAMI_VALUE);
108  if (SENSOR_ERROR_NONE != status)
109  {
110  PRINTF("\r\n Sensor Initialization Failed\r\n");
111  return -1;
112  }
113  PRINTF("\r\n Successfully Initiliazed Sensor\r\n");
114 
115  /*! Set the task to be executed while waiting for I2C transactions to complete. */
116  MAG3110_I2C_SetIdleTask(&mag3110Driver, (registeridlefunction_t)SMC_SetPowerModeVlpr, SMC);
117 
118  /*! Configure the MAG3110 sensor driver. */
119  status = MAG3110_I2C_Configure(&mag3110Driver, cMag3110ConfigNormal);
120  if (SENSOR_ERROR_NONE != status)
121  {
122  PRINTF("\r\n MAG3110 Sensor Configuration Failed, Err = %d\r\n", status);
123  return -1;
124  }
125  PRINTF("\r\n Successfully Applied MAG3110 Sensor Configuration\r\n");
126 
127  for (;;) /* Forever loop */
128  {
129  /*! Wait for data ready from the FXAS21002. */
130  status = MAG3110_I2C_ReadData(&mag3110Driver, cMag3110Status, &dataReady);
131  if (0 == (dataReady & MAG3110_DR_STATUS_ZYXDR_MASK))
132  { /* Loop, if new sample is not available. */
133  continue;
134  }
135 
136  /*! Read the raw sensor data from the FXAS21002. */
137  status = MAG3110_I2C_ReadData(&mag3110Driver, cMag3110OutputNormal, data);
138  if (ARM_DRIVER_OK != status)
139  {
140  PRINTF("\r\n Read Failed. \r\n");
141  return -1;
142  }
143 
144  /*! Process the sample and convert the raw sensor data to signed 16-bit container. */
145  rawData.mag[0] = ((int16_t)data[0] << 8) | data[1];
146  rawData.mag[1] = ((int16_t)data[2] << 8) | data[3];
147  rawData.mag[2] = ((int16_t)data[4] << 8) | data[5];
148 
149  MAG3110_CalibrateHardIronOffset(&rawData.mag[0], &rawData.mag[1], &rawData.mag[2]);
150 
151  /* NOTE: PRINTF is relatively expensive in terms of CPU time, specially when used with-in execution loop. */
152  PRINTF("\r\n Mag X = %d Y = %d Z = %d\r\n", rawData.mag[0], rawData.mag[1], rawData.mag[2]);
153  ASK_USER_TO_RESUME(100); /* Ask for user input after processing 10 samples. */
154  }
155 }
