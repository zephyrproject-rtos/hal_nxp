47 /***********************************************************************

49  * Function Name : imxsdk_gpio_pin_init
50  * Description : Initialize particular GPIO pin used by board.
51  *
52  ***************************************************************************/
53 void imxsdk_gpio_pin_init(
54  pinID_t aPinId, gpio_direction_t dir, void *apPinConfig, gpio_isr_handler_t aIsrHandler, void *apUserData)
55 {
56  gpioConfigiMXSDK_t *pGpioConfig = (gpioConfigiMXSDK_t *)apPinConfig;
57  gpioHandleiMXSDK_t *pinHandle = (gpioHandleiMXSDK_t *)aPinId;
58  if (NULL == apPinConfig)
59  {
60  pGpioConfig = &gpioConfigDefault;
61  }
62 
63  pGpioConfig->pinConfig.direction = dir == GPIO_DIRECTION_IN ? kGPIO_DigitalInput : kGPIO_DigitalOutput;
64  /* Set the pin information */
65  GPIO_PinInit(pinHandle->base, pinHandle->pinNumber, &pGpioConfig->pinConfig);
66 
67  /* Isr is installed */
68  if (aIsrHandler)
69  {
70  /* Enable the IRQ */
71  EnableIRQ(pinHandle->irq);
72  isrObj[pinHandle->portNumber][pinHandle->pinNumber].isrHandle = aIsrHandler;
73  isrObj[pinHandle->portNumber][pinHandle->pinNumber].pUserData = apUserData;
74  /* Enable GPIO pin interrupt */
75  GPIO_PortEnableInterrupts(pinHandle->base, pinHandle->mask);
76  }
77 }

150 /***********************************************************************
151  *
152  * Function Name : imxsdk_gpio_handle_interrupt
153  * Description : handle the gpio interrupt in a pin.
154  *
155  ***************************************************************************/
156 void imxsdk_gpio_handle_interrupt(GPIO_Type *apBase, port_number_t gpioPortNumber)
157 {
158  uint32_t isfr = GPIO_GetPinsInterruptFlags(apBase);
159 
160  /* Parse through all the pending interrupt for a PORT */
161  for (uint8_t i = 0; i < GPIO_NUMBER_OF_PIN; i++)
162  {
163  if (isfr & (1 << i))
164  {
165  gpio_isr_handler_t handle = isrObj[gpioPortNumber][i].isrHandle;
166  if (handle == NULL)
167  {
168  continue;
169  }
170  /* Call user defined handler */
171  handle(isrObj[gpioPortNumber][i].pUserData);
172  GPIO_ClearPinsInterruptFlags(apBase, (1 << i));
173  }
174  }
175 }
