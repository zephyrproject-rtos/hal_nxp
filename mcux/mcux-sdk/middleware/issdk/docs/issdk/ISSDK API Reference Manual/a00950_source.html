134 /*! -----------------------------------------------------------------------
135  * @brief This is the The main function implementation.
136  * @details This function invokes board initializes routines, then then brings up the sensor and
137  * finally enters an endless loop to continuously read available samples.
138  * @param[in] void This is no input parameter.
139  * @return void There is no return value.
140  * @constraints None
141  * @reeentrant No
142  * -----------------------------------------------------------------------*/
143 int main(void)
144 {
145  int32_t status;
146  bool motionDetect;
147  uint16_t lastReportedSteps;
148  uint8_t intStatus, data[FXOS8700_ACCEL_DATA_SIZE];
149 
150  pedometer_t pedometer; /* This handle holds the current configuration and status value for the pedometer.*/
151  fxos8700_accelmagdata_t rawData;
152  fxos8700_i2c_sensorhandle_t fxos8700Driver;
153 
154  ARM_DRIVER_I2C *I2Cdrv = &I2C_S_DRIVER; // Now using the shield.h value!!!
155  GENERIC_DRIVER_GPIO *pGpioDriver = &Driver_GPIO_KSDK;
156 
157  /*! Initialize the MCU hardware. */
158  BOARD_InitPins();
159  BOARD_BootClockRUN();
160  BOARD_InitDebugConsole();
161 
162  PRINTF("\r\n ISSDK FXOS8700 sensor driver example for Motion Activated Pedometer.\r\n");
163 
164  /*! Initialize FXOS8700 pin used by FRDM board */
165  pGpioDriver->pin_init(&FXOS8700_INT1, GPIO_DIRECTION_IN, NULL, &fxos8700_int_callback, NULL);
166 
167  /*! Initialize RGB LED pin used by FRDM board */
168  pGpioDriver->pin_init(&GREEN_LED, GPIO_DIRECTION_OUT, NULL, NULL, NULL);
169 
170  /*! Initialize the I2C driver. */
171  status = I2Cdrv->Initialize(I2C_S_SIGNAL_EVENT);
172  if (ARM_DRIVER_OK != status)
173  {
174  PRINTF("\r\n I2C Initialization Failed\r\n");
175  return -1;
176  }
177 
178  /*! Set the I2C Power mode. */
179  status = I2Cdrv->PowerControl(ARM_POWER_FULL);
180  if (ARM_DRIVER_OK != status)
181  {
182  PRINTF("\r\n I2C Power Mode setting Failed\r\n");
183  return -1;
184  }
185 
186  /*! Set the I2C bus speed. */
187  status = I2Cdrv->Control(ARM_I2C_BUS_SPEED, ARM_I2C_BUS_SPEED_FAST);
188  if (ARM_DRIVER_OK != status)
189  {
190  PRINTF("\r\n I2C Control Mode setting Failed\r\n");
191  return -1;
192  }
193 
194  /*! Initialize FXOS8700 sensor driver. */
195  status = FXOS8700_I2C_Initialize(&fxos8700Driver, &I2C_S_DRIVER, I2C_S_DEVICE_INDEX, FXOS8700_I2C_ADDR,
196  FXOS8700_WHO_AM_I_PROD_VALUE);
197  if (SENSOR_ERROR_NONE != status)
198  {
199  PRINTF("\r\n Sensor Initialization Failed\r\n");
200  return -1;
201  }
202 
203  /*! Set the task to be executed while waiting for I2C transactions to complete. */
204  FXOS8700_I2C_SetIdleTask(&fxos8700Driver, (registeridlefunction_t)SMC_SetPowerModeVlpr, SMC);
205 
206  /* Apply FXOS8700 Configuration for Motion Activated Pedometer. */
207  status = Sensor_I2C_Write(fxos8700Driver.pCommDrv, &fxos8700Driver.deviceInfo,
208  fxos8700Driver.slaveAddress, cFxos8700CommonConfiguration);
209  if (SENSOR_ERROR_NONE != status)
210  {
211  PRINTF("\r\n Write FXOS8700 Initialization Failed.\r\n");
212  return -1;
213  }
214 
215  /*! Initialize the pedometer*/
216  pedometer_init(&pedometer);
217  PRINTF("\r\n Pedometer successfully Initialized and Ready for measurements.");
218 
219  for (;;) /* Forever loop for Motion Detection */
220  {
221  /*! Configure the pedometer*/
222  pedometer_configure(&pedometer, &cPedoConfig);
223 
224  /* Apply FXOS8700 Configuration for Motion Detection. */
225  status = FXOS8700_I2C_Configure(&fxos8700Driver, cFxos8700ConfigMotionDetect);
226  if (SENSOR_ERROR_NONE != status)
227  {
228  PRINTF("\r\n Write FXOS8700 Motion Configuration Failed!\r\n");
229  return -1;
230  }
231 
232  motionDetect = true;
233  pGpioDriver->set_pin(&GREEN_LED);
234  PRINTF("\r\n\r\n Waiting for Motion | MCU switching to Sleep Mode ...\r\n");
235 
236  for (;;) /* Loop for Motion Detection */
237  { /* In ISR Mode we do not need to check Data Ready Register.
238  * The receipt of interrupt will indicate event is ready. */
239  if (false == gFxos8700EventReady)
240  { /* Loop, if new sample is not available. */
241  SMC_SetPowerModeWait(SMC);
242  continue;
243  }
244  else
245  { /*! Clear the data ready flag, it will be set again by the ISR. */
246  gFxos8700EventReady = false;
247  }
248 
249  /*! Read the INT_SRC from the FXOS8700. */
250  status = FXOS8700_I2C_ReadData(&fxos8700Driver, cFxos8700INTSrc, &intStatus);
251  if (ARM_DRIVER_OK != status)
252  {
253  PRINTF("\r\n Read INT SRC Failed!\r\n");
254  return -1;
255  }
256 
257  if (motionDetect)
258  {
259  if ((intStatus & FXOS8700_INT_SOURCE_SRC_FFMT_MASK) == FXOS8700_INT_SOURCE_SRC_FFMT_MASK)
260  {
261  /*! Display that a Motion event has been detected. */
262  PRINTF(" Motion detected...\r\n");
263 
264  /* Apply FXOS8700 Configuration for ASLP Detection with DATA. */
265  status = FXOS8700_I2C_Configure(&fxos8700Driver, cFxos8700ConfigDataReady);
266  if (SENSOR_ERROR_NONE != status)
267  {
268  PRINTF("\r\n Write FXOS8700 DRDY Configuration Failed.\r\n");
269  return -1;
270  }
271  motionDetect = false;
272  pGpioDriver->clr_pin(&GREEN_LED);
273  }
274  }
275  else
276  {
277  /*! Read the Output from the FXOS8700. */
278  status = FXOS8700_I2C_ReadData(&fxos8700Driver, cFxos8700Output, data);
279  if (ARM_DRIVER_OK != status)
280  {
281  PRINTF("\r\n ERROR : Read Data Failed!\r\n");
282  return -1;
283  }
284 
285  /*! Convert the raw sensor data for feeding to pedometer algorithm. */
286  rawData.accel[0] = ((int16_t)data[0] << 8) | data[1];
287  rawData.accel[1] = ((int16_t)data[2] << 8) | data[3];
288  rawData.accel[2] = ((int16_t)data[4] << 8) | data[5];
289 
290  /*! Execute the pedometer Algorithm */
291  pedometer_run(&pedometer, (ped_accel_t *)&rawData.accel);
292 
293  /* Display Steps when there is a change */
294  if (pedometer.status.stepcount && pedometer.status.stepcount != lastReportedSteps)
295  {
296  pGpioDriver->toggle_pin(&GREEN_LED);
297  lastReportedSteps = pedometer.status.stepcount;
298  if(pedometer.status.stepcount == 1)
299  {
300  PRINTF("\r\n | Steps | Distance | Speed | Calories | Activity |\r\n");
301  }
302  PRINTF("\033[K | %5d | %4dm | %2dkmph | %3d | %s |\r",
303  pedometer.status.stepcount, pedometer.status.distance, pedometer.status.speed/1000, pedometer.status.calories, pActivity[pedometer.status.status.bits.activity]);
304  }
305 
306  if ((intStatus & FXOS8700_INT_SOURCE_SRC_ASLP_MASK) == FXOS8700_INT_SOURCE_SRC_ASLP_MASK)
307  {
308  break;
309  }
310  }
311  }
312  }
313 }
