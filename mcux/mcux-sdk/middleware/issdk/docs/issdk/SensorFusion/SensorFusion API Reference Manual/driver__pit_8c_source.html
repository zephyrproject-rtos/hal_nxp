31 /*! \file driver_pit.c
32  \brief Provides a simple abstraction for a periodic interval timer.
33 
34  Bare metal implementations of the sensor fusion library require at least
35  one periodic interrupt for use as a timebase for sensor fusion functions.
36  The Periodic Interval Timer (PIT) is one such module that is commonly
37  found on NXP Kinetis MCUs. The PIT functions are only referenced at the
38  main() level. There is no interaction within the fusion routines themselves.
39 */

62 void pit_init(uint32_t microseconds)
63 {
64  /* initialize PIT */
65  pit_config_t pitConfig; // Declare KSDK PIT configuration structure
66  PIT_GetDefaultConfig(&pitConfig); // Fill out that structure with defaults
67  PIT_Init(PIT, &pitConfig); // PIT is declared by the KSDK
68  // We choose to use Channel 0 of the PIT. That can obviously be changed
69  PIT_SetTimerPeriod(PIT, kPIT_Chnl_0, USEC_TO_COUNT(microseconds, PIT_SOURCE_CLOCK));
70  PIT_EnableInterrupts(PIT, kPIT_Chnl_0, kPIT_TimerInterruptEnable);
71  EnableIRQ(PIT_IRQ_ID);
72  PIT_StartTimer(PIT, kPIT_Chnl_0);
73 }
